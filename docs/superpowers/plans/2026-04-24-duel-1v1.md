> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a new `duel` flag-game mode with two-player simultaneous play, shared rounds, first-tap resolution, dedicated setup/play/results routes, and rematch support without regressing `solo`, `family`, or `explorer`.

**Architecture:** Extend the existing `flag-game` module rather than branching off into a second module. **Tech Stack:** React 19, TypeScript strict mode, React Router, Zustand with Immer, CSS Modules, Vitest, React Testing Library.

```ts
recordDuelAnswer: (playerId, chosenFlag) =>
  set((state) => {
    if (
      state.mode !== 'duel' ||
      state.currentFlag === null ||
      state.difficulty === null ||
      state.duelResolution !== null
    ) {
      return;
    }

    const answeringPlayer = state.duelPlayers.find((player) => player.id === playerId);
    if (!answeringPlayer) return;

    const opponent = getOpponent(state.duelPlayers, playerId);
    const correct = chosenFlag.name === state.currentFlag.name;
    const winnerId = correct ? playerId : opponent.id;
    const loserId = correct ? opponent.id : playerId;
    const resolution: DuelRoundResolution = correct ? 'correct' : 'opponent-awarded';
    const points = DIFFICULTY[state.difficulty].points;

    state.duelScores[winnerId] = (state.duelScores[winnerId] ?? 0) + points;
    state.duelAnsweringPlayerId = playerId;
    state.duelSelectedFlag = chosenFlag;
    state.duelResolvedBy = winnerId;
    state.duelResolution = resolution;
    state.duelHistory.push({
      flag: state.currentFlag,
      winnerId,
      loserId,
      resolution,
      answeringPlayerId: playerId,
    });
  }),

recordDuelTimeout: () =>
  set((state) => {
    if (state.mode !== 'duel' || state.currentFlag === null || state.duelResolution !== null) return;

    state.duelAnsweringPlayerId = null;
    state.duelSelectedFlag = null;
    state.duelResolvedBy = null;
    state.duelResolution = 'timeout';
    state.duelHistory.push({
      flag: state.currentFlag,
      winnerId: null,
      loserId: null,
      resolution: 'timeout',
      answeringPlayerId: null,
    });
  }),

advanceDuelRound: () =>
  set((state) => {

    state.options = [];
    state.duelAnsweringPlayerId = null;

    state.duelResolvedBy = null;
    state.duelResolution = null;
```

```ts
// src/modules/flag-game/data/rounds.ts
import { DIFFICULTY } from './constants';
import { FLAGS } from './flags';
import { pickRandom, shuffle } from './utils';

import type { Continent } from '@/shared/types';
import type { DifficultyKey, Flag } from '../types';

function getPool(maxTier: number, continent: Continent | 'Todos'): Flag[] {
  const baseFlags =
    continent === 'Todos' ? FLAGS : FLAGS.filter((flag) => flag.continent === continent);

  return baseFlags.filter((flag) => flag.tier <= maxTier);
}

export function buildNextRound(
  difficulty: DifficultyKey,
  continent: Continent | 'Todos',
  usedFlags: string[],
): { flag: Flag; options: Flag[] } {
  const diff = DIFFICULTY[difficulty];
  const pool = getPool(diff.maxTier, continent);
  const availableFlags = pool.filter((flag) => !usedFlags.includes(flag.name));
  const candidateFlags = availableFlags.length >= diff.options ? availableFlags : pool;
  const flag = candidateFlags[Math.floor(Math.random() * candidateFlags.length)];
  const wrongOptions = pickRandom(pool.length >= diff.options ? pool : FLAGS, diff.options - 1, [
    flag,
  ]);

  return {
    flag,
    options: shuffle([flag, ...wrongOptions]),
  };
}
```

```ts
export interface DuelSetupState {
  canStart: boolean;

  playerNames: [string, string];
  setDifficulty: (difficulty: DifficultyKey) => void;
  setPlayerName: (index: 0 | 1, value: string) => void;
  start: () => void;
}

function buildPlayers(playerNames: [string, string]): Player[] {
  return playerNames.map((name, index) => ({
    id: `player-${index}`,
    name: name.trim(),
    color: PCOLORS[index],
    avatar: PAVATARS[index],
  }));
}

export function useDuelSetup(): DuelSetupState {

  const startDuel = useGameStore((state) => state.startDuel);
  const [difficulty, setDifficulty] = useState<DifficultyKey>('easy');
  const [playerNames, setPlayerNames] = useState<[string, string]>(['', '']);

  const canStart = playerNames.every((name) => name.trim().length > 0);

  return {
    canStart,
    difficulty,
    playerNames,
    setDifficulty,
    setPlayerName: (index, value) =>
      setPlayerNames((previous) => {
        const next: [string, string] = [...previous] as [string, string];
        next[index] = value;
        return next;
      }),
    start: () => {
      if (!canStart) return;
      startDuel(difficulty, buildPlayers(playerNames));
      navigate('/flag-game/duel/play');
    },
  };
}
```

```ts
// src/modules/flag-game/hooks/useDuelPlayingState.ts
import { useCallback, useEffect, useMemo } from 'react';
import { useNavigate } from 'react-router-dom';

import { ANSWER_DELAY_MS, DEFAULT_ROUND_SECONDS, DIFFICULTY, DUEL_R } from '../data/constants';
import { buildNextRound } from '../data/rounds';
import { useGameStore } from '../store/gameStore';
import { useSettingsStore } from '../store/settingsStore';
import type { Flag, Player } from '../types';

import { useGameSfx } from './useGameSfx';
import { useTimer } from './useTimer';
import { useVisualEffects } from './useVisualEffects';

interface DuelPlayerPanelState {
  feedbackText: string | null;
  onAnswer: (flag: Flag) => void;
  player: Player;
  score: number;
  selected: Flag | null;
}

export interface DuelPlayingState {
  currentFlag: Flag | null;
  options: Flag[];
  playerPanels: DuelPlayerPanelState[];
  ready: boolean;
  roundLabel: string;
  timeLeft: number;
  timerColor: string;
  timerPct: number;
  timerUrgent: boolean;
  visualEffects: ReturnType<typeof useVisualEffects>;
}

function getTimerColor(timerPct: number): string {
  if (timerPct > 50) return '#22c55e';
  if (timerPct > 25) return '#eab308';
  return '#ef4444';
}

function getPlayerFeedback(
  playerId: string,
  resolution: 'correct' | 'opponent-awarded' | 'timeout' | null,
  answeringPlayerId: string | null,
  resolvedBy: string | null,
): string | null {
  if (resolution === null) return null;
  if (resolution === 'timeout') return "Time's up";
  if (playerId === resolvedBy) return 'Correct';
  if (playerId === answeringPlayerId && resolution === 'opponent-awarded') return 'Gifted points';
  return 'Too slow';
}

function getDisplayedSelection(
  playerId: string,
  currentFlag: Flag | null,
  answeringPlayerId: string | null,
  resolvedBy: string | null,
  resolution: 'correct' | 'opponent-awarded' | 'timeout' | null,
  selectedFlag: Flag | null,
): Flag | null {
  if (currentFlag === null || resolution === null || resolution === 'timeout') return null;
  if (playerId === answeringPlayerId) return selectedFlag;
  if (playerId === resolvedBy) return currentFlag;
  return currentFlag;
}

export function useDuelPlayingState(): DuelPlayingState {
  const navigate = useNavigate();
  const sfx = useGameSfx();
  const visualEffects = useVisualEffects();
  const { continent } = useSettingsStore();
  const {
    currentFlag,
    options,
    difficulty,
    usedFlags,
    duelPlayers,
    duelScores,
    duelRound,
    duelResolution,
    duelResolvedBy,
    duelAnsweringPlayerId,
    duelSelectedFlag,
    setRoundData,
    recordDuelAnswer,
    recordDuelTimeout,
    advanceDuelRound,
  } = useGameStore();

  useEffect(() => {
    if (!difficulty || duelPlayers.length !== 2 || currentFlag) return;

    const nextRound = buildNextRound(difficulty, continent, usedFlags);
    setRoundData(nextRound.flag, nextRound.options);
  }, [continent, currentFlag, difficulty, duelPlayers.length, setRoundData, usedFlags]);

  const seconds = difficulty ? DIFFICULTY[difficulty].time : DEFAULT_ROUND_SECONDS;
  const { timeLeft } = useTimer({
    seconds,
    active: !!currentFlag && duelResolution === null,
    onTick: (remainingSeconds) => {
      if (remainingSeconds <= 5 && remainingSeconds > 2) sfx('tick');
      if (remainingSeconds <= 2) sfx('tickUrgent');
    },
    onExpire: () => {
      sfx('timeout');
      recordDuelTimeout();
    },
  });

  useEffect(() => {
    if (duelResolution === null) return;

    const timeoutId = window.setTimeout(() => {
      if (duelRound + 1 >= DUEL_R) {
        sfx('victory');
        navigate('/flag-game/duel/results');
        return;
      }

      advanceDuelRound();
    }, ANSWER_DELAY_MS);

    return () => window.clearTimeout(timeoutId);
  }, [advanceDuelRound, duelResolution, duelRound, navigate, sfx]);

  const playerPanels = useMemo(
    () =>
      duelPlayers.map((player) => ({
        feedbackText: getPlayerFeedback(
          player.id,
          duelResolution,
          duelAnsweringPlayerId,
          duelResolvedBy,
        ),
        onAnswer: (flag: Flag) => {
          if (useGameStore.getState().duelResolution !== null) return;
          sfx(flag.name === currentFlag?.name ? 'correct' : 'wrong');
          recordDuelAnswer(player.id, flag);
        },
        player,
        score: duelScores[player.id] ?? 0,
        selected: getDisplayedSelection(
          player.id,
          currentFlag,
          duelAnsweringPlayerId,
          duelResolvedBy,
          duelResolution,
          duelSelectedFlag,
        ),
      })),
    [
      currentFlag,
      duelAnsweringPlayerId,
      duelPlayers,
      duelResolvedBy,
      duelResolution,
      duelScores,
      duelSelectedFlag,
      recordDuelAnswer,
      sfx,
    ],
  );

  const timerPct = (timeLeft / seconds) * 100;

  return {
    currentFlag,
    options,
    playerPanels,
    ready: duelPlayers.length === 2 && currentFlag !== null,
    roundLabel: `${duelRound + 1}/${DUEL_R}`,
    timeLeft,
    timerColor: getTimerColor(timerPct),
    timerPct,
    timerUrgent: timeLeft <= 5 && duelResolution === null,
    visualEffects,
  };
}
```

```tsx
// src/modules/flag-game/screens/DuelResultsScreen.tsx
import React from 'react';
import { useNavigate } from 'react-router-dom';

import { useGameStore } from '../store/gameStore';

import styles from './DuelResultsScreen.module.css';

export function DuelResultsScreen(): React.JSX.Element {
  const navigate = useNavigate();
  const { difficulty, duelHistory, duelPlayers, duelScores, startDuel } = useGameStore();

  const rankedPlayers = [...duelPlayers].sort(
    (first, second) => (duelScores[second.id] ?? 0) - (duelScores[first.id] ?? 0),
  );
  const winner = rankedPlayers[0] ?? null;
  const second = rankedPlayers[1] ?? null;
  const isTie =
    winner !== null &&
    second !== null &&
    (duelScores[winner.id] ?? 0) === (duelScores[second.id] ?? 0);

  function handleRematch(): void {
    if (!difficulty || duelPlayers.length !== 2) return;
    startDuel(difficulty, duelPlayers);
    navigate('/flag-game/duel/play');
  }

  if (!winner) {
    return <div className={styles.emptyState}>No hay resultados del duelo.</div>;
  }

  return (
    <div className={styles.screen}>
      <div className={styles.hero}>{isTie ? '🤝' : '🏆'}</div>
      <h2 className={styles.title}>{isTie ? '¡Empate!' : `¡${winner.name} gana!`}</h2>
      <div className={styles.scoreRow}>
        {rankedPlayers.map((player) => (
          <div className={styles.scoreCard} key={player.id}>
            <div className={styles.scoreAvatar}>{player.avatar}</div>
            <div className={styles.scoreName}>{player.name}</div>
            <div className={styles.scoreValue}>{duelScores[player.id] ?? 0} pts</div>
          </div>
        ))}
      </div>
      <div className={styles.historyCard}>
        {duelHistory.map((round, index) => (
          <div className={styles.historyRow} key={`${round.flag.name}-${index}`}>
            <span className={styles.historyRound}>R{index + 1}</span>
            <span className={styles.historyFlag}>{round.flag.code}</span>
            <span className={styles.historyName}>{round.flag.name}</span>
            <span className={styles.historyResolution}>{round.resolution}</span>
          </div>
        ))}
      </div>
      <div className={styles.actions}>
        <button className={['btn', styles.primaryButton].join(' ')} onClick={handleRematch} type="button">
          🔄 Revancha
        </button>
        <button
          className={['btn', styles.secondaryButton].join(' ')}
          onClick={() => navigate('/flag-game')}
          type="button"
        >
          🏠 Menú
        </button>
      </div>
    </div>
  );
}
```
